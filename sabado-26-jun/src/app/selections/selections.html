Operaciónes después del select

# selection.append(name) 

			 	var svg = d3.select('svg').append('circle')

# selection.attr(name[, value]) 

				circle = svg.select('circle')
							.attr('r', 20)

# selection.style(name[, value[, priority]]) 

				circle.style('fill','red');

# selection.classed(name[, value]) 

				circle.classed('alecs',true);

Usando el selectAll

# selectAll.(selector) 

			 	circle = svg.selectAll('circle')
							.attr('r', function(i,d){
								return i*10+10; 
							})

Transiciones

# select.transition 

			 	circle = svg.selectAll('circle')
			 				.style("fill", "white")
							.transition()
							.delay(500)
							.duration(5000)
			 				.style("fill", "black")
			 				.attr('r', function(i,d){
								return i*10+10; 
							});

Operaciones con el mouse

# click 

			 	circle = svg.selectAll('circle')
							.on('click', function(d,i){
								alert('Clicked!')
							})

# mouseover

			 	circle = svg.selectAll('circle')
			 				.on('mouseover', function(d,i){
			 					d3.select(this)
								      .style("fill", 'blue');
							})

# mouseout

			 	circle = svg.selectAll('circle')
			 				.on('mouseout', function(d,i){
			 					d3.select(this)
								      .style("fill", 'black');
							});
